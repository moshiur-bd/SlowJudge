#include<bits/stdc++.h>
using namespace std;

int main()
{

#ifdef akid

#endif // akid

    clock_t st=clock();
    int t;
    scanf("%d",&t);
    assert(1<=t and t<=50);
    while(t-->0)
    {
        int A,B;
        scanf("%d %d",&A,&B);

        assert( (1582<=A and A<=10000) and (1582<=B and B<=10000));

        int ans=0;
        for(int i=A; i<=B; i++)
        {
            if((i%4==0 and i%100!=0) or i%400==0)
                ans++;
        }

        printf("%d\n",ans);
    }

    clock_t ed=clock();
    double    time_used = ((double) (ed - st)) /(double) CLOCKS_PER_SEC;
    cerr<<"Time used: "<<time_used<<endl;

    return 0;
}

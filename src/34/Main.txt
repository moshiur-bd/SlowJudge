#include<bits/stdc++.h>
using namespace std;

int main()
{

    int tst;
    long long int n,ball;
    clock_t start, end;
    double time_used;
    start = clock();
    //freopen ( "judge.in", "r", stdin );
    //freopen("judge.out", "w", stdout);
    scanf("%d",&tst);
    assert(tst<=500);

    for(int i=1;i<=tst;i++)
    {
        scanf("%lld",&n);
        ball = n * (n-1) / 2;
        printf("Case %d: %lld\n",i,ball);
    }
    end = clock();
    time_used = ((double) (end - start)) / CLOCKS_PER_SEC;
    //printf("%lf\n",time_used);
    return 0;
}
